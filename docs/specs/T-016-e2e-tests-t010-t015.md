def test_crisis_page_renders(auth_page, live_server):

def test_analyze_page_renders(auth_page, live_server):
    resp = auth_page.goto(f"{live_server}/analyze")
    assert resp.status == 200
    assert auth_page.locator("input[name=ticker]").is_visible()

def test_analyze_invalid_ticker_shows_error(auth_page, live_server):
    auth_page.goto(f"{live_server}/analyze")
    auth_page.fill("input[name=ticker]", "XYZXYZ123FAKE")
    auth_page.click("button[type=submit]")
    auth_page.wait_for_load_state("networkidle")
    assert "Internal Server Error" not in auth_page.content()
    content = auth_page.content()
    assert "не найден" in content or "Тикер" in content

def test_scenario_api_returns_json(auth_page, live_server):
    import json
    # Use CDP to fetch directly via JS
    result = auth_page.evaluate("""
        fetch('/api/scenario?budget=200&return_pct=8&years=20')
          .then(r => r.json())
    """)
    # result is a dict with fv/delta/goal_pct
    assert "fv" in result
    assert "delta" in result
    assert "goal_pct" in result
    assert result["fv"] > 0

### test_hacker.py — новые adversarial тесты
```python
def test_analyze_unauthenticated_redirects(live_server):
    r = httpx.get(f"{live_server}/analyze", follow_redirects=False, timeout=5)
    assert r.status_code in (302, 307)
    assert "login" in r.headers.get("location", "").lower()

def test_crisis_unauthenticated_redirects(live_server):
    r = httpx.get(f"{live_server}/crisis", follow_redirects=False, timeout=5)
    assert r.status_code in (302, 307)

def test_scenario_unauthenticated_redirects(live_server):
    r = httpx.get(f"{live_server}/api/scenario?budget=200&return_pct=8&years=20",
                  follow_redirects=False, timeout=5)
    assert r.status_code in (302, 307)

def test_analyze_xss_ticker_rejected(live_server):
    with _client(live_server) as c:
        r = c.post("/analyze", data={"ticker": "<script>alert(1)</script>"})
    assert r.status_code != 500
    if r.status_code == 200:
        assert "<script>alert(1)</script>" not in r.text

def test_analyze_long_ticker_no_500(live_server):
    with _client(live_server) as c:
        r = c.post("/analyze", data={"ticker": "A" * 50})
    assert r.status_code != 500

def test_analyze_sql_injection_ticker(live_server):
    with _client(live_server) as c:
        r = c.post("/analyze", data={"ticker": "'; DROP TABLE positions; --"})
    assert r.status_code != 500
    assert "error" not in r.text.lower() or "traceback" not in r.text.lower()

def test_scenario_budget_over_limit(live_server):
    with _client(live_server) as c:
        r = c.get("/api/scenario?budget=200000&return_pct=8&years=20")
    assert r.status_code == 422  # FastAPI ge/le validation

def test_scenario_years_over_limit(live_server):
    with _client(live_server) as c:
        r = c.get("/api/scenario?budget=200&return_pct=8&years=100")
    assert r.status_code == 422

def test_scenario_negative_budget_no_500(live_server):
    with _client(live_server) as c:
        r = c.get("/api/scenario?budget=-100&return_pct=8&years=20")
    assert r.status_code in (422, 200)
    assert r.status_code != 500

def test_scenario_zero_budget_returns_result(live_server):
    with _client(live_server) as c:
        r = c.get("/api/scenario?budget=0&return_pct=8&years=20")
    assert r.status_code == 200
```

## 11. Red Flags
- `test_scenario_api_returns_json` использует `page.evaluate()` с fetch — требует что auth cookie передаётся в JS context (auth_page уже имеет cookie)
- `test_analyze_invalid_ticker_shows_error` может быть медленным (yfinance timeout ~5s)
- При пустой БД `test_tooltips_present_on_recommend` может не найти `.term` если страница пустая — нужна проверка через page source, не через visible()
